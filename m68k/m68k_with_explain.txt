.data
    .org     0x000
addr_in_port:      .word  128  ; порт ввода
addr_out_port:     .word  132  ; порт вывода

    .text
    .org     0x088

_start:                        ; инициализация стека и запуск прогаммы
    movea.l  4092, A7          ; Копирование 4092 в A7 => Инициализирует указатель стека (A7/SP) безопасным адресом в конце памяти. Без этого нельзя вызывать процедуры. 
    jsr      main              ; Переход к основной функции (адрес возврата кладется в стек).
    halt                       ; Остановка симулятора после возврата из main.

main:                          ; инциализация всех адресов
    movea.l  2048, A0          ; A0 = Указатель на буфер, где будет храниться исходная строчка 
    movea.l  addr_in_port, A1  ; A1 = Загружает адрес метки 0x80
    movea.l  (A1), A1          ; A1 = Разыменовывает его, получая фактический адрес порта (128).
    movea.l  addr_out_port, A5 ; То же самое для порта вывода 0x84
    movea.l  (A5), A5          ; A5 = Фактический адрес порта вывода (132).
    movea.l  2500, A3          ; A3 = Указатель на буфер итогового результата.

    movea.l  3000, A6          ; A6 = Базовый адрес для таблицы (заполняется в обратном порядке).
    move.l   1, D0             ; D0 = Текущая степень (начинаем с 10^0 = 1).
    move.l   10, D2            ; D2 = Множитель (10).
    move.l   10, D3            ; D3 = Счетчик цикла (нам нужно 10 чисел).    

init_pow_loop:                 ; заполнение таблички 10 зачем то (думаю что для логического и)
    move.l   D0, -(A6)         ; Пред-декремент: смещаем A6 на 4 байта назад и пишем туда D0.
    mul.l    D2, D0            ; Умножаем D0 на 10.
    sub.l    1, D3             ; Уменьшаем счетчик D3 на 1.
    bne      init_pow_loop     ; Если D3 не ноль, повторяем цикл, так как флаги поменялись на прошлой иструкции все работет хорошо

    clr.l    D1                ; Очищаем счетчик длины прочитанной строки.

read_fmt:                      ; Считываем строку для форматирования по адресу в A0
    cmp.l    33, D1            ; Сравнение чиселок. Ограничение: защита от переполнения буфера (макс 33 символа).
    beq      error_exit        ; Переход если Zero=1 => Если превысили — выход с ошибкой.
    clr.l    D0                ; Очищаем регистр для нового символа.
    move.b   (A1), D0          ; Читаем 1 байт из порта ввода (A1).
    cmp.l    10, D0            ; Сравниваем символ с 10 (ASCII код переноса строки '\n').
    beq      fmt_done          ; Если это '\n', формат прочитан полностью.
    move.b   D0, (A0)+         ; Сохраняем символ в буфер формата (A0) и сдвигаем указатель на 4 байта.
    add.l    1, D1             ; Увеличиваем счетчик длины.
    jmp      read_fmt          ; Замыкаем цикл.

fmt_done:                      ; концовка цикла считывания
    clr.b    (A0)              ; Записываем нуль-терминатор в конец строки формата. Необходимо в С строках
    movea.l  2048, A0          ; Сбрасываем указатель A0 обратно на начало буфера формата.   

scan_loop:                     ; Парсинг строки и форматирования. Этот блок ищет спецификатор %d и извлекает его параметры
    clr.l    D0                ; Очистка D0              
    move.b   (A0)+, D0         ; Читаем символ из буфера формата в D0.
    beq      program_exit      ; Если дошли до нуль-терминатора (0), конец работы.
    cmp.l    37, D0            ; Это знак процента '%' (ASCII 37)?
    beq      handle_format     ; Если да, начинаем разбор формата.
    move.b   D0, (A3)+         ; Иначе — просто копируем символ в итоговый буфер (A3).
    jmp      scan_loop         ; Замыкаем цикл.

handle_format:                 ; проверяет флаг левого выравнивания: если следующий символ 45 (знак -), в ячейку памяти 3600 записывается 1.
    clr.l    D6                ; D6 будет хранить ширину поля.
    movea.l  3600, A4          ; A4 = Адрес флага выравнивания (по левому/правому краю).
    clr.l    (A4)              ; По умолчанию флаг = 0 (выравнивание вправо).

    clr.l    D0                ; очистка D0
    move.b   (A0), D0          ; Смотрим следующий символ после '%'.
    cmp.l    45, D0            ; Это знак минуса '-' (ASCII 45)? Выравнивание влево
    bne      hf_loop           ; Если нет, переходим к чтению ширины.
    move.b   (A0)+, D0         ; Если минус есть, "съедаем" его.
    move.l   1, (A4)           ; Записываем 1 во флаг выравнивания (влево).

hf_loop:                       ; парсит ширину отступа (цифры)
    clr.l    D0                ; очистка D0
    move.b   (A0)+, D0         ; Читаем символ ширины.
    cmp.l    100, D0           ; Это буква 'd' (ASCII 100)?
    beq      print_integer     ; Формат собран, идем печатать число.
    
    sub.l    48, D0            ; Превращаем ASCII код в реальную цифру (вычитаем код '0'). Для числа 5: 53 - 48 = 5
    bmi      scan_loop         ; Если символ меньше '0', формат некорректен, прерываем.
    cmp.l    10, D0            ; Проверка: цифра должна быть от 0 до 9.
    bpl      scan_loop         
    
    move.l   10, D3            ; Подготовка множителя 10
    mul.l    D3, D6            ; Шаг 3: Умножаем то, что уже успели собрать, на 10
    add.l    D0, D6            ; Шаг 4: Прибавляем новую цифру к результату
    jmp      hf_loop           ; Повторяем для следующего символа

print_integer:                 ; Этап перевода 32-битного бинарного числа (полученного из read_int) в ASCII строку.
    jsr      read_int          ; Вызываем процедуру чтения числа из ввода. Результат в D0.
    movea.l  3500, A2          ; A2 = Временный буфер для формирования строки числа.
    clr.l    D7                ; D7 = Длина получившегося числа.
    
    cmp.l    0, D0             ; Число отрицательное?
    bpl      pi_buf_pos        ; Если положительное (Branch Plus), перепрыгиваем.
    clr.l    D3
    sub.l    D0, D3            ; Делаем число положительным (0 - D0).
    move.l   D3, D0
    move.b   45, (A2)+         ; Пишем знак '-' во временный буфер.
    add.l    1, D7             ; Увеличиваем счетчик длины числа.

    cmp.l    0, D0             ; Особенность 32-бит: -2147483648 при модуле остается отрицательным.
    bpl      pi_buf_pos
    move.b   50, (A2)+         ; Если это тот самый крайний случай, жестко пишем '2' в буфер.
    add.l    1, D7
    move.l   1, D5             ; Ставим флаг "не печатать ведущие нули".
    sub.l    2000000000, D0    ; Отрезаем 2 миллиарда, оставляем 147483648.
    movea.l  2964, A6          ; Указываем таблицу степеней без 10^9.
    move.l   9, D4             ; Будем обрабатывать только 9 разрядов.
    jmp      pi_buf_loop

pi_buf_pos:                    ; Обработка числа
    movea.l  2960, A6          ; Стандартная таблица степеней (от 10^9).
    move.l   10, D4            ; Обрабатываем все 10 разрядов.
    clr.l    D5                ; Флаг ведущих нулей выключен.

pi_buf_loop:                   ; Внешний цикл по разрядам числа.
    move.l   (A6)+, D1         ; Достаем текущую степень 10 (например, 1 000 000).
    move.l   48, D3            ; D3 = код '0' (начальное значение для цифры).

pi_buf_sub:                    ; Внутренний цикл: "деление" вычитанием. Мы постепенно уменьшаем разрядность получая итоговое число на кажом разряде
    cmp.l    D1, D0            ; Если число меньше степени...
    bmi      pi_buf_digit_done ; ...разряд посчитан.
    sub.l    D1, D0            ; Иначе вычитаем степень из числа.
    add.l    1, D3             ; Увеличиваем цифру (ASCII код) на 1.
    jmp      pi_buf_sub

pi_buf_digit_done:             ; Обработка получившегося числа, а точнее цифры из нашего числа
    cmp.l    48, D3            ; Это ноль?
    bne      pi_buf_write      ; Если не ноль, печатаем.
    cmp.l    0, D5             ; Если ноль, были ли до этого значащие цифры?
    bne      pi_buf_write      ; Если да, печатаем.
    cmp.l    1, D4             ; Если это самый последний разряд (единицы), печатаем ноль всегда.
    beq      pi_buf_write
    jmp      pi_buf_next       ; Иначе пропускаем (убираем ведущие нули).

pi_buf_write:                  ; Сохраняем вычисленную цифру в буфер
    move.l   1, D5             ; Взводим флаг "была значащая цифра". Как только мы встретили первую цифру, отличную от нуля, мы записываем 1 в D5, чтобы последующие нули (например, в числе 105) не отсекались как "ведущие".
    move.b   D3, (A2)+         ; Копируем ASCII-код цифры из D3 во временный буфер числа (A2 = 3500) и сдвигаем курсор A2 на 1 байт вперед.
    add.l    1, D7             ; Увеличиваем счетчик фактической длины получившейся строки числа (D7) на 1.

pi_buf_next:                   ; Сколько свободного места осталось в рамках заданной пользователем ширины.
    sub.l    1, D4             ; Уменьшаем счетчик разрядов (D4 изначально равен 10 для всех степеней десятки).
    bne      pi_buf_loop       ; Если обработали еще не все 10 разрядов (D4 != 0), прыгаем в начало цикла конвертации на следующую степень десятки.

    move.l   D6, D3            ; Копируем запрошенную ширину поля из D6 (которую мы распарсили в hf_loop) в рабочий регистр D3.
    sub.l    D7, D3            ; Вычитаем из запрошенной ширины (D3) реальную длину нашего числа (D7). Результат — количество необходимых пробелов.
    bpl      pi_check_align    ; Если результат положительный или равен нулю (Branch Plus), значит, ширина поля больше числа. Идем проверять выравнивание.
    clr.l    D3                ; Если результат отрицательный (число оказалось длиннее, чем указанная ширина, например, число "12345" при формате %2d), то отступы не нужны. Сбрасываем D3 в 0.

pi_check_align:                ; Начало выравнивания
    movea.l  3600, A4          ; Загружаем в адресный регистр A4 адрес ячейки памяти 3600, где хранится флаг выравнивания.
    move.l   (A4), D1          ; Читаем значение флага из памяти в регистр D1.
    cmp.l    1, D1             ; Проверяем, равен ли флаг единице?
    beq      pi_left_align     ; Если равен 1 (был знак минус, например %-5d), то прыгаем на блок левого выравнивания. Иначе — идем строго вниз на правое.

pi_right_align:                ; Выравнивание вправо (число слева от пробелов)
    cmp.l    0, D3             ; Проверяем, нужно ли вообще добавлять пробелы (D3 == 0?)
    beq      pi_ra_num         ; Если пробелов делать не нужно, сразу прыгаем к копированию числа.

pi_ra_pad_loop:                ; Запись пробельчиков
    move.b   32, (A3)+         ; Записываем ASCII-код пробела (32) в итоговый буфер (A3 = 2500) и смещаем курсор.
    sub.l    1, D3             ; Уменьшаем счетчик нужных пробелов на 1.
    bne      pi_ra_pad_loop    ; Крутим цикл, пока не запишем все пробелы (пока D3 != 0).

pi_ra_num:                     ; Концовка выравнивания справа
    movea.l  3500, A2          ; Сбрасываем указатель A2 на самое начало временного буфера (3500), где лежит наше готовое ASCII-число.
    cmp.l    0, D7             ; Проверяем длину числа. Защита: если длина почему-то 0, то копировать нечего.
    beq      pi_done           ; Выходим из блока.

pi_ra_num_loop:                ; Запись числа
    move.b   (A2)+, (A3)+      ; Читает 1 байт (ASCII-символ цифры) из временного буфера по адресу в A2 (3500) и записывает его в итоговый буфер по адресу в A3 (2500). 
    sub.l    1, D7             ; Уменьшает счетчик оставшейся длины числа в регистре D7 на единицу.
    bne      pi_ra_num_loop    ; Проверяет флаг Zero (Z). Если D7 еще не равен нулю, возвращается в начало цикла pi_ra_num_loop, чтобы скопировать следующую цифру.

pi_left_align:                 ; Выравнивание влево
    movea.l  3500, A2          ; Загружает адрес 3500 (начало временного буфера собранного числа) в адресный регистр A2. Это сброс указателя, так как на этапе конвертации он ушел вперед.
    cmp.l    0, D7             ; Сравнивает длину получившегося числа в D7 с нулем.
    beq      pi_la_pad         ; Если длина числа равна нулю (Branch Equal), то копировать нечего. Программа сразу прыгает к заполнению поля пробелами в pi_la_pad.

pi_la_num_loop:                ; цикл проверки символов
    move.b   (A2)+, (A3)+      ; Побайтово копирует ASCII-символы числа из временного буфера (A2) в итоговый буфер (A3). 
    sub.l    1, D7             ; Уменьшает счетчик длины числа в D7 на 1.
    bne      pi_la_num_loop    ; Если в D7 еще остались символы (D7 != 0), замыкает цикл, возвращаясь на pi_la_num_loop.

pi_la_pad:                     ; Нужны ли пробелы
    cmp.l    0, D3             ; Сравнивает оставшееся количество необходимых пробелов в D3 с нулем.
    beq      pi_done           ; Если пробелы не нужны (D3 == 0, то есть ширина формата меньше или равна длине числа), прыгает на завершение обработки формата в pi_done.

pi_la_pad_loop:                ; Заполняем пробелами
    move.b   32, (A3)+         ; Записывает ASCII-код пробела (число 32) в итоговый буфер (A3) и смещает указатель A3 на следующий байт. Число как бы "добивается" пробелами справа.
    sub.l    1, D3             ; Уменьшает счетчик необходимых пробелов в D3 на 1.
    bne      pi_la_pad_loop    ; Если пробелы еще нужны (D3 != 0), продолжает цикл записи пробелов.

pi_done:                       ; Завершение обработки текущего спецификатора
    jmp      scan_loop         ; Программа возвращается к чтению строки формата (ищет следующие символы или новый знак %).

program_exit:                  ; Выходим из программы
    move.b   0, (A3)           ; Записывает байт 0 (нуль-терминатор) в текущую позицию итогового буфера, куда указывает A3. Это критически важно, чтобы обозначить жесткий конец собранной строки.
    movea.l  2500, A2          ; Загружает адрес 2500 (самое начало нашего итогового буфера, где лежит готовый текст с числами и пробелами) в адресный регистр A2.  

pe_loop:                       ; Вывод символов в 0x84
    clr.l    D0                ; Полностью очищает регистр данных D0 перед чтением.
    move.b   (A2)+, D0         ; Читает один ASCII-символ из итогового буфера (A2) в регистр D0 и сдвигает указатель памяти A2 на 1 байт вперед.
    beq      pe_done           ; Если это наш нуль-терминатор — вся строка напечатана, прыгаем на pe_done.
    move.b   D0, (A5)          ; Пересылает ASCII-символ из D0 на адрес порта вывода (A5 содержит 132 (0x84)).
    jmp      pe_loop           ; Переходит в начало цикла pe_loop для отправки на экран следующего символа строки.

pe_done:                       ; Возврат из главной программы
    rts                        ; Завершает выполнение главной функции main и возвращает управление в стартовый блок к команде halt.

rror_exit:                     ; Краш выход aka sys.exit(0)
    clr.l    D0                ; Полностью очищает регистр данных D0 (делает его равным 0).
    sub.l    1, D0             ; Вычитает 1 из нуля. В компьютерной знаковой арифметике (дополнительный код) регистр D0 заполняется единицами и превращается в число -1 (0xFFFFFFFF).
    move.l   D0, (A5)          ; Выбрасывает все 32 бита этого маркера ошибки (-1) целиком в порт вывода (A5 = 132). Симулятор перехватывает это число как сигнал системной аварии.
    halt                       ; hlt

read_int:                      ; Циклы считывания числа для форматирования
    clr.l    D0                ; D0 = Аккумулятор итогового числа.
    clr.l    D2                ; D2 = Флаг отрицательного числа (1 = минус).
    clr.l    D4                ; D4 = Счетчик валидных цифр.
    
    clr.l    D1                ; Очистка главного регистра для работы
    move.b   (A1), D1          ; Чтение первого символа из адреса ввода
    cmp.l    45, D1            ; Это минус?
    bne      ri_check_digit    ; Если нет, проверяем как цифру.
    move.l   1, D2             ; Если да, ставим флаг минуса.
    clr.l    D1
    move.b   (A1), D1          ; Читаем следующий символ.  

ri_check_digit:                ; После того, как считали знак минуса если он вообще был, переходим сюда
    cmp.l    48, D1            ; Проверка на '0' 
    bmi      ri_invalid        ; Число кал друзья
    cmp.l    58, D1            ; Проверка на верхнюю границу ('9' + 1) т. е меньше 10
    bpl      ri_invalid        ; Число кал друзья
    
    sub.l    48, D1            ; Превращаем в цифру.
    add.l    D1, D0            ; Добавляем первую цифру в аккумулятор D0.
    add.l    1, D4             ; Увеличиваем счетчик валидных цифр.

ri_loop:
    clr.l    D1                ; Очистка главного регистра для работы
    move.b   (A1), D1          ; Читаем следующий символ.
    cmp.l    10, D1            ; Это перенос строки '\n'?
    beq      ri_done           ; Если да, парсинг окончен.
    
    cmp.l    48, D1            ; Проверка на '0' 
    bmi      ri_invalid        ; Число кал друзья
    cmp.l    58, D1            ; Проверка на верхнюю границу ('9' + 1) т. е меньше 10
    bpl      ri_invalid        ; Число кал друзья
    
    sub.l    48, D1            ; Превращаем в цифру.
    add.l    1, D4             ; Увеличиваем счетчик валидных цифр.
    
    cmp.l    0, D0             ; Проверяем текущее накопленное в аккумуляторе D0 число.
    bmi      ri_invalid        ; Если D0 стал отрицательным на этом этапе, значит переполнение УЖЕ произошло  

    cmp.l    214748364, D0     ; Аккумулятор уже равен INT_MAX / 10?
    bmi      ri_multiply       ; Если меньше, безопасно умножать на 10.
    beq      ri_check_exact    ; Если равен, нужно проверить последнюю вводимую цифру.
    jmp      ri_invalid        ; Если больше, число точно переполнит 32 бита.       

ri_check_exact:                ; Проверка последнего числа на переполнение
    cmp.l    1, D2             ; Проверяем регистр-флаг знака D2.
    beq      ri_check_neg_overflow ; Если там лежит 1 (число отрицательное), то лимиты другие. Прыгаем на ветку отриц переполнения.

    cmp.l    7, D1             ; Сравниваем новую считанную цифру в D1 с критической семеркой.
    bmi      ri_multiply       ; Если новая цифра МЕНЬШЕ 7 (то есть от 0 до 6), то итоговое число будет <= 2147483646. Это безопасно, прыгаем на умножение.
    beq      ri_multiply       ; Если новая цифра РАВНА 7, то итоговое число станет ровно 2147483647 (INT_MAX). Это пиковое значение, но оно валидно. Тоже на умножение.
    jmp      ri_invalid        ; Если цифра БОЛЬШЕ 7 (8 или 9), то число станет 2147483648+, что вызовет переполнение. Аварийный выход.

ri_check_neg_overflow:         ; проверка переполнения отрицательного числа
    cmp.l    8, D1             ; Сравниваем новую цифру в D1 с критической восьмеркой.
    bmi      ri_multiply       ; Если цифра МЕНЬШЕ 8 (0..7), то модуль числа будет <= 2147483647. Это безопасно, идем на умножение.
    beq      ri_multiply       ; Если цифра РАВНА 8, то модуль станет ровно 2147483648 (а само число -2147483648). Это нижний предел INT_MIN. Валидно, идем на умножение.
    jmp      ri_invalid        ; Если цифра РАВНА 9 (число -2147483649), то разрядность превышена. Аварийный выход.

ri_multiply:                   ; Умножаем число на 10 и записываем туда наше число
    move.l   10, D3            ; Записываем константу 10 в регистр D3. Она используется как множитель для сдвига.
    mul.l    D3, D0            ; Умножаем текущее накопленное число в D0 на 10 (например, было 4, стало 40).
    add.l    D1, D0            ; Прибавляем к полученному числу новую цифру из D1 (например, 40 + 2 = 42).
    jmp      ri_loop           ; Безусловный прыжок обратно в начало цикла ri_loop для чтения следующего символа.

ri_invalid:                    ; Точка выхода из программы если число неверное
    jmp      error_exit        ; Прыжок на выход с ошибкой

ri_done:                       ; Задача этого блока — сделать число отрицательным, если в самом начале ввода был обнаружен знак «минус».
    cmp.l    0, D4             ; Проверяем регистр D4 (счетчик прочитанных цифр).
    beq      ri_invalid        ; Если мы прочитали 0 цифр (например, пользователь просто нажал Enter или ввел только знак минус) — это ошибка, падаем.

    cmp.l    1, D2             ; Проверяем флаг знака в D2.
    bne      ri_exit           ; Если флаг НЕ равен 1 (число положительное), то менять ничего не надо. Сразу прыгаем на выход.

    clr.l    D1                ; Если число должно быть отрицательным. Очищаем D1 (делаем его равным 0).
    sub.l    D0, D1            ; Вычитаем наше накопленное положительное число из нуля (0 - D0) и сохраняем разность в D1.
    move.l   D1, D0            ; Переносим обратно: теперь в основном регистре D0 лежит правильное отрицательное число.

ri_exit:                       ; Возвращает процессор обратно в функцию print_integer
    rts                        ; Инструкция выталкивает адрес возврата из вершины стека (A7)
