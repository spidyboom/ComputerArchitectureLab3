_start:  \ СтаРТуЕМ!!
    -1 0 >r  \ кладем -1 на стек => кладем 0 на стек => записываем 0 в стек возврата

read_str_loop:  \ записываем буквочки на стек
    @p 0x80 dup 10 xor  \ чиатем адресс 0x80 кладем на стек => дублируем => кладем 10 (\n) на стек => xor T и S
    if hit_nl  \ если на стеке 0 (символ 10) прыгаем
    dup 0 xor  \ дублируем => кладем 0 (\0) на стек => xor T и S
    if hit_null  \ если на стеке 0 (символ 0) прыгаем
    r> 1 + dup >r 32 xor  \ стек возвратов на дата стек => 1 на стек => слагаем => кладем обратно в стек возврата => 32 на стек => xor T и S
    if overflow  \ если на стеке 0 (счетчик 32) прыгаем 
    read_str_loop ;  \ прыжок выше (лупа)

hit_nl:  \ устанавливаем в A ноль (указание на буффер) этот кусок кода для обработки \n
    drop 0 a!  \ удаляем десятку со стека => 0 на стек => запись стека в A

flush_nl_loop:  \ цикл считывания со стека и записи в память и в 0x84
    dup inv if flush_nl_done  \ дублируем стек => инвертируем => если ноль прыгаем (проверка что мы на дне стека)
    dup !p 0x84 !+  \ копируем число на стеке => сохраняем верхушку стека в 0x84 => сохраняем верхушку стека в адрес A также A+1
    flush_nl_loop ;  \ прыжок выше (лупа)

flush_nl_done:  \ очищаем стек и ставим в A нолик
    drop 0 !+ r> drop  \ дропаем -1 со стека => записываем 0 на стек => записываем его в адрес из A также A+1 => верхушка возврата в дата стек => снова дроп
    pad_buffer halt  \ вызываем подпрограмму => останавливаем 

hit_null:  \ устанавливаем в A ноль (указание на буффер) этот кусок кода для обработки \0
    drop r> 1 + dup >r 32 xor if overflow  \ дроп нолика со стека => возврат записываем в дату => +1 => дату в возврат => проверка переполнения
    0 a!  \ записываем 0 на стек => записываем этот 0 в A

flush_null_loop: \ цикл считывания со стека и записи в память и в 0x84
    dup inv if flush_null_done  \ дублируем => инвертация => если 0 прыгаем (проверка что мы на дне стека)
    dup !p 0x84 !+  \ копируем число на стеке => сохраняем верхушку стека в 0x84 => сохраняем верхушку стека в адрес A также A+1
    flush_null_loop ;  \ прыжок выше (лупа)

flush_null_done:  \ очистка стека и запись 0 в память 
    drop 0 !+  \ дропаем значение мракера -1 => записываем 0 на стек => сохраняем верхушку стека в адрес A также A+1

rest_loop:  \ читаем то что было после нолика записывая это в память
    @p 0x80 dup 10 xor  \ чиатем адресс 0x80 кладем на стек => дублируем => кладем 10 (\n) на стек => xor T и S
    if rest_nl  \ если на стеке 0 (символ не 10) прыгаем
    r> 1 + dup >r 32 xor if overflow  \ стек возвратов на дата стек => 1 на стек => слагаем => 32 на стек => xor T и S => если на стеке 0 (счетчик не 32) прыгаем 
    !+ rest_loop ;  \ сохраняем верхушку стека в адрес A также A+1 => прыжок выше (лупа)

rest_nl:  \ завершение счета, очистка и дозаполнение
    drop 0 !+ r> drop  \ дропаем стек => записываем 0 на стек => записываем его в адрес из A также A+1 => верхушка возврата в дата стек => снова дроп
    pad_buffer halt  \ вызываем подпрограмму => останавливаем 

overflow:  \ обработка ошибок или переполнений
    -858993460 !p 0x84 halt  \ 0xCCC нас стек => запись в 0x84 => остановка

pad_buffer:  \ подпрограмма дозаполнения буффера
pad_loop:  \ заполняем оставшуюся часть стека нижними подчеркиваниями
    a 32 xor if pad_finish  \ копируем a на стек => 32 на стек => xor T и S => если конец прыжок на выход из подпрограммы
    0x5F !+ pad_loop ;  \ записываем "_" на стек => записываем его в адрес из A также A+1 => прыжок выше (лупа)
pad_finish: ;  \ выход из подпрограммы
