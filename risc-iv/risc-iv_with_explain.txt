.text

_start:
    ; записываем 1 в верхние 20 бит sp => sp = 0x0001000
    ; это наш указатель стека 4096
    lui  sp, 1
    
    ; копирование 4 байтов из 0x80 в a0
    ; наше число на проверку
    lw   a0, 128(zero)

    ; сохраняем адрес следющей строки в ra и переходим на count_ones
    ; вызываем функцию и сохраняем адрес следующей строки
    jal  ra, count_ones

    ; копируем финальный результат в ячейку 132
    sw   a0, 132(zero)
    ; записывает 0 в a0
    addi a0, zero, 0
    ; выход
    halt

count_ones:
    ; вычитаем 8 байт из нзначения стека - выделяем две ячейки по 32 бита
    addi sp, sp, -8
    ; копирует значение ra (наш указатель на count_ones) в стек + 4
    ; сохраняем адрес возврата дабы не заблудится
    sw   ra, 4(sp)

    ; сравниваем значение с zero регистром, если ноль прыгает вниз
    ; Наше число подошло к завершению жаль
    beqz a0, _ones_zero

    ; побитовое и регистра a0 и 1 => в t0
    ; проверяем самый крайний бит нашего слова
    andi t0, a0, 1
    ; копирует t0 в sp+0
    ; копируем в память sp результат проверки
    sw   t0, 0(sp)
    
    ; сдвигает биты в a0 на один шаг вправо
    ; отбрасываем уже проверенный бит
    srli a0, a0, 1
    ; сохраняем адрес следющей строки в ra и переходим на count_ones
    ; вызываем функцию и сохраняем адрес следующей строки
    jal  ra, count_ones
    
    ; блок далее будет суммироввать то что осталось после выполнения прохода
    ; копируем данные из sp+0 в t0
    lw   t0, 0(sp)
    ; складываем регистр a0 и t0
    add  a0, a0, t0
    ; jmp на _ones_exit
    j    _ones_exit

_ones_zero:
    ; записывает 0 в a0
    ; конец рекурсии a0 стал нулем
    addi a0, zero, 0

_ones_exit:
    ; копируем значение из памяти в ra
    ; достаем адрес по которому нам надо поднятся выше
    lw   ra, 4(sp)
    ; прибавляем 8 к указатель стека
    addi sp, sp, 8
    ; jmp в адрес указанный в ra
    jr   ra
